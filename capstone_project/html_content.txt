      console.warn('Failed to parse last checkout items.')
    }

    if(gtmEvent) {
      gtmEvent.event = 'purchase';
      localStorage.removeItem('summmer_lastBeginCheckout');
    } else {
      gtmEvent = {
        event: "purchase",
        ecommerce: {
          currency: currency,
          value: 0,
          items: []
        }
      };
    }
    
    pushEventToDataLayer(gtmEvent);
  }
